def sarake_oikein(sudoku: list, rivi: int):
	rivi2 = []
	for i in range(len(sudoku)):
		if sudoku[i][rivi] > 0:
			rivi2.append(sudoku[i][rivi])
		if len(rivi2) != len(set(rivi2)):
			return False
		else:
			return True
							
def nelio_oikein(sudoku: list, rivi_nro: int, sarake_nro: int):
	nelio = []
	nelio2 = []
	for i in range(3):
		rivi = sudoku[rivi_nro]
		nelio.append(rivi[sarake_nro:sarake_nro + 3])
		rivi_nro += 1
		for alkio in nelio:
			for nro in alkio:
				if nro > 0:
					nelio2.append(nro)
		if len(nelio2) != len(set(nelio2)):
				return False
			else:
				return True
							
def sudoku_oikein(sudoku: list):
	riviO = True
	sarakeO = True
	nelioO = True
	nelio_tarkastus = [[0, 0], [0, 3], [0, 6], [3, 0], [3, 3], [3, 6], [6, 0], [6, 3], [6, 6]]
	for i in range(len(sudoku)):
		if rivi_oikein(sudoku, i) == False:
			riviO = False
			break
	for i in range(len(sudoku)):
		if sarake_oikein(sudoku, i) == False:
			sarakeO = False
			break
	for i in range(len(nelio_tarkastus)):
		if nelio_oikein(sudoku, nelio_tarkastus[i][0], nelio_tarkastus[i][1]) == False:
			nelioO = False
			break
							
	if riviO == True and sarakeO == True and nelioO == True:
		return True
	else:
		return False
